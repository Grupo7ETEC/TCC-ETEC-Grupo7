CREATE TABLE `formulario` (
  `id` int NOT NULL AUTO_INCREMENT,
  `nome` varchar(80) DEFAULT NULL,
  `endereco` varchar(80) DEFAULT NULL,
  `cep` varchar(10) DEFAULT NULL,
  `email` varchar(80) DEFAULT NULL,
  `tel` varchar(11) DEFAULT NULL,
  `idsegunda` varchar(10) DEFAULT NULL,
  `idterca` varchar(10) DEFAULT NULL,
  `idquarta` varchar(10) DEFAULT NULL,
  `idquinta` varchar(10) DEFAULT NULL,
  `idsexta` varchar(10) DEFAULT NULL,
  `idsabado` varchar(10) DEFAULT NULL,
  `s_email` varchar(10) DEFAULT NULL,
  `hr` varchar(100) DEFAULT NULL,
  `nome_animal` varchar(40) DEFAULT NULL,
  `gato` varchar(10) DEFAULT NULL,
  `cachorro` varchar(10) DEFAULT NULL,
  `duvida` varchar(40) DEFAULT NULL,
  `ajuda` varchar(200) DEFAULT NULL,
  `sim` varchar(5) DEFAULT NULL,
  `nao` varchar(5) DEFAULT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci
